Sub 時間単位をnsからsに変換して､累積時間分を無視()
    ''【説明】：時間単位をnsからsに変換して､累積時間分を無視
    ''累積時間[s]の列を追加するコード部分
    '列を追加
    Columns(2).Insert
    
    '列をコピー
    Columns(1).Copy Destination:=Columns(2)
    
    '列の名称を変更
    Cells(1, 2) = "累積時間[s]"
    
    '末尾の行数を取得＆定義
    Dim end_row_num As Integer: end_row_num = Cells(1, 2).End(xlDown).Row
    
    '累乗のスケールを定義
    Dim scale_exponentiation As Integer: scale_exponentiation = -9
    
    '累積時間[s]の列をループ計算
    For i = 2 To end_row_num
    Cells(i, 2) = Cells(i, 2) * 10 ^ scale_exponentiation
    Next
    
    ''時間[s]の列を追加するコード部分
    '列を追加
    Columns(3).Insert
    
    '列をコピー
    Columns(2).Copy Destination:=Columns(3)
    
    '列の名称を変更
    Cells(1, 3) = "時間[s]"
        
    '時間[s]の列をループ計算
    For j = 2 To end_row_num
    Cells(j, 3) = Cells(j, 2) - Cells(2, 2)
    Next
End Sub
